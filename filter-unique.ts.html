export const filterUnique = (arr: Array<any>) => {
  return arr.filter((item, index) => {
    const value = (item as string).trim();
    return value !== '' && arr.map((item: string) => item.trim()).indexOf(value) === index;
  });
};
 
 
export const filterUniqueObjects: <T extends { [key: string]: any }>(
  arr: Array<T>,
  key: string
) => Array<T> = (arr, key) => {
  return arr.filter((item, index) => {
    return arr.map((item) => item[key] as string).indexOf(item[key]) === index;
  });
};
